# Character profile with films, starships, and homeworld
query charProfile {
  person(id: "cGVvcGxlOjEw") {
    name
    filmConnection {
      films {
        title
      }
    }
    starshipConnection {
      starships {
        name
      }
    }
    homeworld {
      name
    }
  }
}

# get first 5 chars, with name and population of homeworld
query charHomes {
  allPeople(first: 5) {
    people {
      name
      homeworld {
        name
        population
      }
    }
  }
}

# for first 3 vehicles, list names, list pilots, list pilot species
query vehiclePilots {
  allVehicles(first: 3) {
    vehicles {
      name
      pilotConnection {
        pilots {
          name
          species {
            name
          }
        }
      }
    }
  }
}

# for first 3 films, list characters, planets, and starships
query filmInfo {
  allFilms(first: 3) {
    films {
      title
      characterConnection {
        characters {
          name
        }
      }
      planetConnection {
        planets {
          name
        }
      }
      starshipConnection {
        starships {
          name
        }
      }
    }
  }
}
